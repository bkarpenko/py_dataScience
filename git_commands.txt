pwd - get current working directory
cd ~/<new directory> - change working directory
ls - cd lss - see files
git remote -v  - chech remote refferences 
git status - discover which file(s) have been changed since the last save
git diff <directory> <filename>- see what changes have been made to the files
git log <directory> <filename> - view the log of the project's history (Press the space bar to go down a page or the 'q' key to quit)




git clone <link on repository> - clone repository on local computer
git add <file name1> <file name2>
git commit -m "<comments on made changes>"
git push origin master - отправить в репозиторий на github (origin) ветку master (на локальном компьютере)
git branch <name of branch> - create branch
git checkout <name of branch> - switch to branch
git branch -m <oldname> <newname> - rename specific branch
git branch <newname> - rename current branch
#checkout to main branch  git merge <brach name to merge with main brand> 
git reset HEAD -- <file> - delete file from staged
 
git rm file1.txt

git commit -m "comment on your action" - save files in staged area and make comment on what was done
